import java.util.*; class Solution { public List<String> summaryRanges(int[] nums) { int left = 0; List<String> result= new ArrayList<>(); String r=""; if(nums.length>0) { for(int i= 1; i<nums.length;i++){ if(nums[i] == nums[i-1]+1){ continue; } if(left !=(i-1)){ r=Integer.toString(nums[left])+"->"+Integer.toString(nums[i-1]); }else{ r=Integer.toString(nums[i-1]); } left = i; result.add(r); r=""; } if(left!=(nums.length-1)){ r=Integer.toString(nums[left])+"->"+Integer.toString(nums[nums.length-1]); }else{ r=Integer.toString(nums[left]); } result.add(r); } return result; } }
